    /**
     * THIS FUNCTION WILL RANDAMIZE THE ARRAYS
     * THIS IS THE MAIN FUNCTION TO RANDOMIZE THE ARRAYS THAT ARE PASSED
     * @PARAM nums, numArr
     **/
    function chooseNum(nums, numArr) {
        if (nums &gt; numArr.length) {
            return false;
        }

        var chooseArr = new Array();
        var tempArr = new Array();
        for (var l_i = 0;l_i &lt; numArr.length;l_i++) {
            tempArr[l_i] = numArr[l_i];
        }

        for (var i = 0;i &lt; nums;i++) {
            chooseArr[chooseArr.length] = tempArr[Math.floor((tempArr.length - 1) * Math.random())];
            var temp = chooseArr[chooseArr.length - 1];
            for (var j = 0;j &lt; tempArr.length;j++) {
                if (tempArr[j] == temp) {
                    tempArr[j] = null;
                    var tempArr2 = new Array();
                    for (var k = 0;k &lt; tempArr.length;k++)
                        if (tempArr[k] != null)
                            tempArr2[tempArr2.length] = tempArr[k];
                    tempArr = tempArr2;
                    break;
                }
            }
        }
        return chooseArr;
    }

    function fetchUserPass() {
        var successStatus = 200;
        var readyStateSuccess = 4;
        var authImgComp = document.getElementById('newDiv');
        var x;
        if (authImgComp != null) {
            x = authImgComp.childNodes[0];
        }
        if (x != null) {
            while (authImgComp.firstChild) {
                authImgComp.removeChild(authImgComp.firstChild);
            }
        }
        if (fLogon()) {
            var xmlhttp;
            if (window.XMLHttpRequest) {
                xmlhttp = new XMLHttpRequest();
            }
            else {
                alert("This browser not supported. Browsers supported : IE7+, Firefox, Chrome, Safari");
                return false;
            }
            xmlhttp.open("POST", "j_security_check", false);
            var paramString = "j_username=" + document.frmLogon.j_username.value + "&amp;j_password=" + document.frmLogon.j_password.value;
            xmlhttp.onreadystatechange = function () {
                if (xmlhttp.readyState == readyStateSuccess) {
                    if (xmlhttp.status == successStatus) {
                        //window.location.href = "../com.ofss.fc.ui.view/faces/main.jspx"
                        document.forms["frmLogon"].submit();
                    }
                    else {
                        var oImg;
                        var txt;
                        var span;

                        oImg = document.createElement("img");
                        oImg.setAttribute('src', 'images/common/qualifier/error_16_qual.png');
                        authImgComp.appendChild(oImg);
                        txt = document.createTextNode("Authentication denied.");
                        span = document.createElement('span');
                        span.className = "AFError"
                        span.appendChild(txt);
                        authImgComp.appendChild(span);
                        oImg = document.createElement("br");
                        authImgComp.appendChild(oImg);
                        txt = document.createTextNode("Invalid Username or Password.");
                        span = document.createElement('span');
                         span.className = "AFError AFSpacer"
                        span.appendChild(txt);
                        authImgComp.appendChild(span);
                    }
                }
            }
            xmlhttp.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
            xmlhttp.send(paramString);
        }
    }
